class Jugador{

    var mana: Int
    
    init(nombre: String, vida: Int, mana: Int) {
        self.nombre = nombre
        self.vida = vida
        self.mana = mana
    }
    
    func información_del_jugador(){
        print("\nNombre del jugador: \(nombre)\nVida: \(vida)\nMana: \(mana)")
    }
}

class Arma {
    var nombre: String
    var tipo: String
    var dañoBase: Int
    var calidad: String
    var dañoFinal: Int
    
    init(dañoBase: Int) {
        self.nombre = Arma.generarArma()
        self.tipo = Arma.definirTipo(nombre: self.nombre)
        self.dañoBase = dañoBase
        self.calidad = Arma.generarCalidad()
        self.dañoFinal = Arma.calcularDaño(base: dañoBase, calidad: self.calidad)
    }
    
    static func generarArma() -> String {
        let armas = ["Uchigatana", "Longinus Spear", "Arco Largo", "Tomo mágico de fuego", "Hacha de batalla", "Ballesta Relampago"]
        return armas.randomElement()!
    }
    
    static func definirTipo(nombre: String) -> String {
        switch nombre {
        case "Uchigatana":
            return "Cuerpo a Cuerpo"
        case "Longinus Spear":
            return "Cuerpo a Cuerpo"
        case "Arco Largo":
            return "Arma a Distancia"
        case "Ballesta Relampago":
            return "Arma a Distancia"
        case "Tomo mágico de fuego":
            return "Arma Mágica"
        case "Hacha de batalla":
            return "Cuerpo a Cuerpo"
        default:
            return "Desconocido"
        }
    }
    
    static func generarCalidad() -> String {
        let prob = Int.random(in: 1...100)
        switch prob {
        case 100:
            return "Legendario"
        case 91...99:
            return "Épico"
        case 76...90:
            return "Raro"
        case 51...75:
            return "Poco Común"
        case 2...50:
            return "Común"
        default:
            return "Roto"
        }
    }
    
    static func calcularDaño(base: Int, calidad: String) -> Int {
        switch calidad {
        case "Legendario":
            return Int(Double(base) * 5.0)
        case "Épico":
            return Int(Double(base) * 2.5)
        case "Raro":
            return Int(Double(base) * 2.0)
        case "Poco Común":
            return Int(Double(base) * 1.5)
        case "Común":
            return base
        case "Roto":
            return Int(Double(base) * 0.5)
        default:
            return base
        }
    }
    
    func información_del_arma() {
        print("\nNombre del arma: \(nombre)\nTipo: \(tipo)\nCalidad: \(calidad)\nDaño: \(dañoFinal)")
    }
}
